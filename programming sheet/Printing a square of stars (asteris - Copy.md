##### Printing a square of stars (asterisks), given the size of its side.



\#include <iostream>

using namespace std;



int main() {

&#x20;   int side;

&#x20;   cout << "Enter the size of the square's side: ";

&#x20;   cin >> side;



&#x20;   for (int i = 0; i < side; i++) {

&#x20;       for (int j = 0; j < side; j++) {

&#x20;           cout << "\* ";

&#x20;       }

&#x20;       cout << endl;

&#x20;   }

&#x20;   return 0;

}



* note :
* المتغير i (الحلقة الخارجية) يتحكم في الصفوف (Rows) (النزول من أعلى لأسفل).
* 
* المتغير j (الحلقة الداخلية) يتحكم في الأعمدة (Columns) (الطباعة من اليسار لليمين داخل نفس السطر).
* 
* لماذا؟
* شاشة الطباعة (Console) في البرمجة تعمل مثلما نكتب على الورق؛ فهي تطبع سطراً كاملاً من اليسار لليمين، ثم تنزل للسطر الذي يليه، ولا يمكنها العودة للأعلى لطباعة عمود.
* 
* لذلك، المنطق البرمجي يسير هكذا:
* 
* نثبت الصف باستخدام المتغير i.
* 
* نستخدم المتغير j لطباعة جميع النجوم في هذا الصف (أي نمر على الأعمدة المجاورة لبعضها).
* 
* عندما تنتهي حلقة j (يكتمل السطر)، نقوم بطباعة سطر جديد cout << endl;.
* 
* تزيد قيمة i لننتقل إلى الصف التالي بالأسفل... وهكذا.
* 
* إليك هذا المخطط التوضيحي الذي يربط المتغيرين i و j بمكان النجمة (بافتراض أن حجم المربع 3):
* 
* Plaintext
* &#x20;              العمود 0      العمود 1      العمود 2
* &#x20;              (j = 0)       (j = 1)       (j = 2)
* &#x20;             -------------------------------------
* الصف 0 (i = 0) |   \*             \*             \*
* الصف 1 (i = 1) |   \*             \*             \*
* الصف 2 (i = 2) |   \*             \*             \*
